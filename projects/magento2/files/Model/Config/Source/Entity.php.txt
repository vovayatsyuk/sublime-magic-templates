class {classname} implements \Magento\Framework\Data\OptionSourceInterface
{{
    /**
     * Options getter
     *
     * @return array
     */
    public function toOptionArray()
    {{
        return [
            ['value' => '$1', 'label' => __('$2')],$0
        ];
    }}

     * Get options in "key-value" format
